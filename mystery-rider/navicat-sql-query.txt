//HTML


SELECT
a.CODE,a.CATEGORY,a.SECTION,a.SUB_SECTION_CODE,a.SUB_SECTION,a.INPUT_TYPE,
b.CHOICE_TEXT, b.MANDATORY 
FROM MR_SURVEY.MR2_SECTIONS a 
LEFT JOIN MR_SURVEY.MR2_CHOICES b ON a.MR2_SECTIONS_ID = b.MR2_SECTIONS_ID 
ORDER BY a.CODE, a.SUB_SECTION_CODE, b.MR2_CHOICES_ID



//TRIGGER.JS

SELECT
a.CODE, a.SUB_SECTION_CODE,a.INPUT_TYPE,
b.CHOICE_TEXT, b.TRIGGER_ACTIVE_IDS, b.TRIGGER_INACTIVE_IDS, b.MANDATORY 
FROM MR_SURVEY.MR2_SECTIONS a 
LEFT JOIN MR_SURVEY.MR2_CHOICES b ON a.MR2_SECTIONS_ID = b.MR2_SECTIONS_ID 
WHERE b.TRIGGER_ACTIVE_IDS is not NULL or b.TRIGGER_INACTIVE_IDS is not NULL or b.MANDATORY is not NULL
ORDER BY a.CODE, a.SUB_SECTION_CODE, a.CODE
